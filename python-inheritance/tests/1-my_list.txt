#This test file contains a test cases for print_sorted function

>>> MyList = __import__('1-my_list').MyList

Basic Sort
----------
>>> my_list = MyList()
>>> my_list.append(5)
>>> my_list.append(7)
>>> my_list.append(4)
>>> my_list.append(3)
>>> my_list.append(8)
>>> my_list.append(2)
>>> my_list.append(6)
>>> my_list.append(1)
>>> my_list.print_sorted()
[1, 2, 3, 4, 5, 6, 7, 8]


Empty List
----------
>>> my_list = MyList()
>>> my_list.print_sorted()
[]


Single Element List
-------------------
>>> my_list = MyList()
>>> my_list.append(6)
>>> my_list.print_sorted()
[6]


Already Sorted List
-------------------
>>> my_list = MyList()
>>> my_list.append(4)
>>> my_list.append(5)
>>> my_list.append(6)
>>> my_list.append(7)
>>> my_list.append(8)
>>> my_list.append(9)
>>> my_list.append(10)
>>> my_list.print_sorted()
[4, 5, 6, 7, 8, 9, 10]


Reverse Sorted List
-------------------
>>> my_list = MyList()
>>> my_list.append(9)
>>> my_list.append(8)
>>> my_list.append(7)
>>> my_list.append(6)
>>> my_list.append(5)
>>> my_list.append(4)
>>> my_list.append(3)
>>> my_list.print_sorted()
[3, 4, 5, 6, 7, 8, 9]


List with Duplicate Elements
----------------------------
>>> my_list = MyList()
>>> my_list.append(2)
>>> my_list.append(4)
>>> my_list.append(2)
>>> my_list.append(3)
>>> my_list.append(3)
>>> my_list.append(7)
>>> my_list.append(8)
>>> my_list.append(9)
>>> my_list.append(5)
>>> my_list.print_sorted()
[2, 2, 3, 3, 4, 5, 7, 8, 9]


Nrgative number
---------------
>>> my_list = MyList()
>>> my_list.append(-5)
>>> my_list.append(0)
>>> my_list.append(-1)
>>> my_list.append(1)
>>> my_list.append(-2)
>>> my_list.print_sorted()
[-5, -2, -1, 0, 1]
